How to extract report_updated_at from Patriots.com

Team injury pages usually include at least one of these:

1) Meta tags (best / most stable)

Try:

meta[property="article:modified_time"]

meta[property="article:published_time"]

meta[name="last-modified"] (less common)

2) A visible “Updated” / date line on the page

Search for text nodes containing “Updated” or a date.

Here’s a robust extraction function you can drop into the scraper:

function parseReportUpdatedAt($) {
  // 1) Try meta tags first
  const metaCandidates = [
    $('meta[property="article:modified_time"]').attr("content"),
    $('meta[property="article:published_time"]').attr("content"),
    $('meta[name="last-modified"]').attr("content"),
  ].filter(Boolean);

  for (const c of metaCandidates) {
    const d = new Date(c);
    if (!isNaN(d)) return d.toISOString();
  }

  // 2) Look for visible "Updated" text
  const bodyText = $("body").text().replace(/\s+/g, " ").trim();
  const m = bodyText.match(/Updated\s+([A-Za-z]{3,9}\s+\d{1,2},\s+\d{4})/i);
  if (m?.[1]) {
    const d = new Date(m[1]);
    if (!isNaN(d)) return d.toISOString();
  }

  // 3) As a last resort, return null (don’t lie in UI)
  return null;
}

function renderInjuryFooter({ report_updated_at, fetched_at, source_url, source_name }) {
  const reportLine = report_updated_at
    ? `Report updated: ${formatShortDate(report_updated_at)}`
    : null;

  const checkedLine = fetched_at
    ? `Last checked: ${formatRelative(fetched_at)}`
    : null;

  return `
    <div class="sc-injury-foot">
      <div class="sc-injury-foot__left">
        ${reportLine ? `<div class="sc-injury-foot__muted">${reportLine}</div>` : ``}
        ${checkedLine ? `<div class="sc-injury-foot__muted">${checkedLine}</div>` : ``}
      </div>
      <a class="sc-injury-card__link" href="${escapeAttr(source_url)}" target="_blank" rel="noopener">
        View Full Injury Report →
      </a>
    </div>
  `;
}

function formatShortDate(iso) {
  const d = new Date(iso);
  return isNaN(d) ? "" : d.toLocaleDateString(undefined, { month:"short", day:"numeric", year:"numeric" });
}

function formatRelative(iso) {
  const d = new Date(iso);
  if (isNaN(d)) return "";
  const diff = Math.floor((Date.now() - d.getTime()) / 1000);
  if (diff < 60) return "just now";
  if (diff < 3600) return `${Math.floor(diff/60)}m ago`;
  if (diff < 86400) return `${Math.floor(diff/3600)}h ago`;
  return `${Math.floor(diff/86400)}d ago`;
}
